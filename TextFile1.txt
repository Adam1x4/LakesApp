using System;

class Program
{
    static void Main()
    {
        int[] numbers = { 10, 5, 8, 15, 3 };
        int maxNumber = numbers[0]; // Assume the first number is the maximum

        for (int i = 1; i < numbers.Length; i++)
        {
            if (numbers[i] > maxNumber)
            {
                maxNumber = numbers[i];
            }
        }

        Console.WriteLine("The maximum number in the array is: " + maxNumber);
    }
}
